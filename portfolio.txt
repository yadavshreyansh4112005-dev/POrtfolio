const codeDisplay = document.getElementById('codeDisplay');
let codeIdx = 0;
function addCodeLine() {
  if (codeIdx >= codeLines.length) {
    codeDisplay.innerHTML = '';
    codeIdx = 0;
  }
  const line = codeLines[codeIdx];
  const div = document.createElement('div');
  div.className = 'code-line';
  div.innerHTML = `<span style="color:rgba(224,230,255,0.3);margin-right:8px;">${String(codeIdx+1).padStart(2,'0')}</span><span style="color:${line.color};">${line.text}</span>`;
  div.style.animationDelay = '0s';
  codeDisplay.appendChild(div);
  codeIdx++;
  // Keep only last 12 lines visible
  while (codeDisplay.children.length > 12) codeDisplay.removeChild(codeDisplay.firstChild);
}
setInterval(addCodeLine, 600);
addCodeLine();

function addChatMsg(text, isUser) {
  const div = document.createElement('div');
  div.className = `chat-msg ${isUser ? 'user' : 'bot'}`;
  div.innerHTML = `<div class="chat-bubble ${isUser ? 'user-bubble' : 'bot-bubble'}">${text}</div>`;
  chatMessages.appendChild(div);
  chatMessages.scrollTop = chatMessages.scrollHeight;
}

function getBotResponse(input) {
  const lower = input.toLowerCase();
  if (lower.includes('skill') || lower.includes('tech') || lower.includes('know')) return botResponses.skills;
  if (lower.includes('project') || lower.includes('work') || lower.includes('built')) return botResponses.projects;
  if (lower.includes('contact') || lower.includes('email') || lower.includes('phone') || lower.includes('reach')) return botResponses.contact;
  if (lower.includes('about') || lower.includes('who') || lower.includes('shreyansh')) return botResponses.about;
  if (lower.includes('ai') || lower.includes('artificial') || lower.includes('automat')) return botResponses.ai;
  if (lower.includes('hi') || lower.includes('hello') || lower.includes('hey')) return botResponses.hello;
  return botResponses.default;
}

function sendChat() {
  const text = chatInput.value.trim();
  if (!text) return;
  addChatMsg(text, true);
  chatInput.value = '';
  setTimeout(() => addChatMsg(getBotResponse(text), false), 600);
}

document.getElementById('chatSendBtn').addEventListener('click', sendChat);
chatInput.addEventListener('keydown', (e) => { if (e.key === 'Enter') { e.preventDefault(); sendChat(); } });
